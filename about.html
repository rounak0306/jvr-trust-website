  <!-- Inline init for Trustee slider -->

  <script>

    (function(){

      function initSlider(sliderId, prevId, nextId, dotsId, interval=5000){

        const slider = document.getElementById(sliderId);

        if(!slider) return;

        const slides = Array.from(slider.querySelectorAll(".slide"));

        const prev = document.getElementById(prevId);

        const next = document.getElementById(nextId);

        const dotsWrap = document.getElementById(dotsId);

        let current = 0, timer = null, width = slider.clientWidth;


        function buildDots(){

          if(!dotsWrap) return [];

          dotsWrap.innerHTML = "";

          return slides.map((_,i)=>{

            const b=document.createElement("button");

            b.setAttribute("role","tab");

            b.setAttribute("aria-label","Go to slide "+(i+1));

            if(i===0) b.classList.add("active");

            b.addEventListener("click", ()=>goTo(i,true));

            dotsWrap.appendChild(b);

            return b;

          });

        }

        let dots = buildDots();


        function updateUI(){

          slides.forEach((s,i)=>s.classList.toggle("active", i===current));

          dots.forEach((d,i)=>d.classList.toggle("active", i===current));

        }

        function goTo(i, user=false){

          current = (i + slides.length) % slides.length;

          const left = current * width;

          slider.scrollTo({ left, behavior: "smooth" });

          updateUI();

          if(user) resetAuto();

        }

        function nextSlide(){ goTo(current+1); }

        function prevSlide(){ goTo(current-1); }

        function startAuto(){ timer = setInterval(nextSlide, interval); }

        function resetAuto(){ clearInterval(timer); startAuto(); }


        window.addEventListener("resize", ()=>{ width = slider.clientWidth; goTo(current); });


        if(prev) prev.addEventListener("click", ()=>{ prevSlide(); resetAuto(); });

        if(next) next.addEventListener("click", ()=>{ nextSlide(); resetAuto(); });


        // touch support

        let sx=0, dx=0;

        slider.addEventListener("touchstart", e=>{ sx=e.touches[0].clientX; dx=0; });

        slider.addEventListener("touchmove", e=>{ dx=e.touches[0].clientX - sx; });

        slider.addEventListener("touchend", ()=>{ if(Math.abs(dx)>50){ dx<0?nextSlide():prevSlide(); resetAuto(); } });


        updateUI();

        startAuto();

      }


      initSlider("trusteeSlider", "trusteePrevBtn", "trusteeNextBtn", "trusteeSliderDots", 4500);

    })();

  </script>
